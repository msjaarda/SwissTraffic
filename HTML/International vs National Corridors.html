Start with a reproduction of N. Maddah Graph

Year = [2003 2005 2007];
SName{1} = 'Ceneri';
SName{2} = 'Trubbach';
SName{3} = 'Mattstetten';
x = 0;
% Set colours
C = linspecer(9);
St = [{'--'} {':'} {'-'} {'--'} {':'} {'-'}];
for i = 1:length(SName)
for k = 1:length(Year)
x = x+1;
try
load(['PrunedS1 WIM/',SName{i},'/',SName{i},'_',num2str(Year(k)),'.mat']);
catch
continue
end
% Let the Classify function add the .CLASS column to PD
PDC = Classify(PD);
% 1. Disqualification by weight (try under 6 or 10 tonnes)
PD = PDC(PDC.GW_TOT > 3500,:);
% 2. Disqualification by Swiss10 Class (exclude 2,3,4,6)
%PD = PD(PD.CS == 1 | PD.CS == 5 | PD.CS == 7 | PD.CS == 8 | PD.CS == 9 | PD.CS == 10,:);
[TotDaysOpen(x), y] = size(unique(PD.JJJJMMTT));
[NumTrucks(x), z] = size(PD);
ADTT(x) = NumTrucks(x)/TotDaysOpen(x);
AvgWeight(x) = mean(PD.GW_TOT)*0.00980665;
StdWeight(x) = std(PD.GW_TOT)*0.00980665;
[nums{x}, edges{x}] = histcounts(PD.GW_TOT*0.00980665,200,'Normalization','pdf');
mids{x} = edges{x}(1:end-1) + diff(edges{x}) / 2;
Between = mids{x}>60 & mids{x}<500;
if i == 3
plot(mids{x}(Between), nums{x}(Between),'-k')
elseif k ~= 3
plot(mids{x}(Between), nums{x}(Between),'Color',C(i,:),'LineStyle',St{k},'HandleVisibility','off')
else
plot(mids{x}(Between), nums{x}(Between),'Color',C(i,:),'LineStyle',St{k})
end
hold on
end
end
title('Vehicle Weight Histogram Comparison')
xlabel('Truck Weight (kN)')
ylabel('PDF')
set(gca,'yticklabel',[])
xlim([0 500])
legend('Ceneri 2003:5:7', 'Trubbach 2003:5:7','Mattstetten 2003');
grid on; hold off
